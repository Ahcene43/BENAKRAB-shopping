function renderOrders(orders) { ordersListEl.innerHTML = ''; if (!orders.length) { ordersListEl.textContent = 'لا توجد طلبات.'; return; } orders.forEach(order => { const orderItem = document.createElement('div'); orderItem.className = 'order-item'; // Préparer valeurs avec fallback const customerName = order.customerName || 'غير محدد'; const phone = order.phone || 'غير محدد'; const address = order.address || 'غير محدد'; const products = order.products || 'غير محدد'; const status = order.status || 'pending'; const dateText = order.createdAt ? new Date(order.createdAt).toLocaleString('ar-EG') : 'غير محدد'; // Utilisation sûre de innerHTML après échappement pour les champs dynamiques orderItem.innerHTML = `

👤 ${escapeHtml(customerName)}

📞 ${escapeHtml(phone)}

📍 ${escapeHtml(address)}

📦 ${escapeHtml(products)}

🔄 الحالة: ${status === 'pending' ? 'قيد الانتظار' : 'مكتمل'}

📅 ${escapeHtml(dateText)}

`; const actions = orderItem.querySelector('.order-actions'); if (status === 'pending') { const completeBtn = document.createElement('button'); completeBtn.textContent = '✅ تم التوصيل'; completeBtn.addEventListener('click', () => updateOrderStatus(order.id, 'completed')); actions.appendChild(completeBtn); } else { const undoBtn = document.createElement('button'); undoBtn.textContent = '↩️ إعادة إلى قيد الانتظار'; undoBtn.addEventListener('click', () => updateOrderStatus(order.id, 'pending')); actions.appendChild(undoBtn); } const deleteBtn = document.createElement('button'); deleteBtn.textContent = '🗑️ حذف الطلب'; deleteBtn.addEventListener('click', () => { if (confirm('هل تريد حقًا حذف هذا الطلب؟')) { firebaseService.deleteOrder(order.id).then(res => { if (res.success) loadDeliveryOrders(); }); } }); actions.appendChild(deleteBtn); ordersListEl.appendChild(orderItem); }); } async function loadDeliveryOrders() { ordersListEl.textContent = 'جاري التحميل...'; const orders = await firebaseService.getDeliveryOrders(); renderOrders(orders); } async function updateOrderStatus(orderId, status) { const res = await firebaseService.updateOrderStatus(orderId, status); if (res.success) { await loadDeliveryOrders(); } else { alert('خطأ عند تحديث حالة الطلب: ' + (res.error || 'Unknown')); } } // Initial load document.addEventListener('DOMContentLoaded', () => { loadDeliveryOrders(); });
